# =============================================================================
# BURO STAAL DASHBOARD - ENVIRONMENT VARIABLES
# =============================================================================
# Kopieer dit bestand naar .env en vul de waarden in

# =============================================================================
# SECURITY - REQUIRED
# =============================================================================

# JWT Secret for authentication (REQUIRED in production)
# Generate: openssl rand -base64 32
# NEVER use the default value in production!
JWT_SECRET=your-generated-jwt-secret-here

# Dashboard Password (REQUIRED)
# This is the password users enter to login to the dashboard
DASHBOARD_PASSWORD=your-secure-dashboard-password

# =============================================================================
# DATABASE - TURSO (REQUIRED)
# =============================================================================

# Turso Database URL (krijg je van Turso dashboard)
TURSO_DATABASE_URL=libsql://your-database-name.turso.io

# Turso Auth Token (krijg je van Turso dashboard)
TURSO_AUTH_TOKEN=your-turso-auth-token-here

# Hoe krijg je Turso credentials?
# 1. Ga naar https://turso.tech
# 2. Maak een gratis account aan
# 3. Maak een nieuwe database aan
# 4. Kopieer de TURSO_DATABASE_URL en TURSO_AUTH_TOKEN
# 5. Run: node setup-turso-db.mjs (om schema aan te maken)
# 6. Run: node seed-turso-db.mjs (optioneel: voor test data)

# =============================================================================
# OPTIONAL - INTEGRATIONS
# =============================================================================

# Slack Webhook (optioneel - voor notificaties bij nieuwe items)
# Krijg je van: https://api.slack.com/messaging/webhooks
SLACK_WEBHOOK=https://hooks.slack.com/services/YOUR/WEBHOOK/URL

# Webflow API (optioneel - indien je Webflow CMS gebruikt)
WEBFLOW_API_HOST=https://api-cdn.webflow.com/v2
WEBFLOW_SITE_API_TOKEN=your-webflow-site-token
WEBFLOW_CMS_SITE_API_TOKEN=your-webflow-cms-token
